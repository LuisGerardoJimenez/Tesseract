	\Battr{numeroIU}{Número: }{Número de la Pantalla. Es un valor numérico entero y este dato es \hyperlink{tRequerido}{requerido} {\em (no se puede omitir)}. Este atributo debe de contener como máximo 20 dígitos. Caracteres admitidos: [0-9] $|$ [.]}
		
	\Battr{nombreIU}{Nombre: }{Nombre que identificará a la Pantalla. Es una frase o enunciado y este dato es \hyperlink{tRequerido}{requerido} {\em (no se puede omitir)}. Este atributo debe de contener como mínimo 10 y máximo 200 caracteres. Caracteres admitidos: [A-Z] $|$ [a-z] $|$ [ñ-á-é-í-ó-ú] $|$ [Ñ-Á-É-Í-Ó-Ú] $|$ [0-9] $|$ [ ] $|$ [-] $|$ [$,$, .]}

Caracteres admitidos: [A-Z] $|$ [a-z] $|$ [ñ-á-é-í-ó-ú] $|$ [Ñ-Á-É-Í-Ó-Ú] $|$ [0-9] $|$ [  ] $|$ [\_] $|$ [-] $|$ [$!$, $?$, \&, @, \%, \#, $($, $)$, ., :, >, <, *, =, $,$]}
